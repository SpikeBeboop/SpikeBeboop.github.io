具体实现#

shortcode#

layouts/shortcodes/details.html 
{{- $summary := .Get "summary" | default (.Get "title") -}}
{{- if not $summary -}}
    {{- warnf "missing value for param 'summary': %s" .Position -}}
{{- end -}}

{{- $open := .Get "open" | default true -}}

<details class="post-details shortcode-details"{{ if $open }} open{{ end }}>
    <summary class="post-details-summary">
        {{- $summary | markdownify -}}
    </summary>
    <div class="post-details-content">
        {{- .Inner | markdownify -}}
    </div>
</details>

assets/css/extended/shortcodes.css 
.shortcode-details {
    margin: 1rem 0;

    cursor: pointer;
